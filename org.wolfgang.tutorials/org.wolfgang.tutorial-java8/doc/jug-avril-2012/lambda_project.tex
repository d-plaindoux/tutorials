\begin{frame}[fragile]{Les Interfaces Fonctionnelles}

\begin{block}{Définition (Interface Fonctionnelle)}
  Interface ayant une et une seule méthode abstraite.
\end{block}

\breakinframe

Définition reposant essentiellement sur:
\begin{enumerate}
\item le type fonctionnel porté par la méthode abstraite
\item aucune convention de nommage de la méthode e.g. {\tt apply}
\end{enumerate}

\begin{alertblock}{}
  \coffeebreak ~ Aucun évolution syntaxique pour le langage des types 
\end{alertblock}

\breakinframe

\begin{exampleblock}{}
\begin{lstlisting}  
  interface Runnable { void run(); }                        
  interface ActionListener { void actionPerformed(ActionEvent e); } 
  interface Comparator<T> { int compare(T t1, T t2); }
  interface Fun<E,R,throws X> { R apply(E e) throws X; }    
\end{lstlisting}
\end{exampleblock}

\end{frame}

%% ========================================

\begin{frame}[fragile]{Extension d'Interface Fonctionnelle}

L'Extension d'Interface Fonctionnelle est Fonctionnelle si et seulement si:
\begin{enumerate}
\item Le nom de la méthode abstraite est identique et
\item Il existe une relation de sous-typage entre les deux spécifications
\begin{itemize}
\item Base formelle à partir des travaux de L. Cardelli, B. Pierce etc.
\item Repose sur les principes de covariance et contravariance 
\item Existe déjà dans les version précédentes
\end{itemize}
\end{enumerate}

\breakinframe \vspace{1em}

\begin{alertblock}{}
  \coffeebreak ~ Unicité de la méthode abstraite vérifiée à la compilation
\end{alertblock}

\end{frame}

\begin{frame}[fragile]{Extension d'Interface et Sous-Typage (Rappel)}

Soit une interface {\tt FunA}
\begin{exampleblock}{}  
\begin{lstlisting}
  interface FunA { A m(B b) throws E1; }
\end{lstlisting}
\end{exampleblock}

et {\tt FunB} une extension de {\tt FunA} 
\begin{exampleblock}{}  
\begin{lstlisting}
  interface FunB extends FunA { C m(D b) throws E2; }
\end{lstlisting}
\end{exampleblock}

\vspace{1em}

On dit alors que {\tt FunB {\color{blue}{extends}} FunA} est valide si et seulement si:
\begin{itemize}
\item {\tt B} est un sous-type de {\tt D}, 
\item {\tt C} est un sous-type de {\tt A} et
\item {\tt E2} est un sous-type de {\tt E1}.
\end{itemize}
  
\end{frame}

\begin{lstlisting}
  interface TA {}
  interface TB extends TA {}

  interface FunBB { TB apply(TB b); }
  interface FunBC extends FunBB { TC apply(TB a); }
  interface FunBA extends FunBB { TA apply(TB a); }
\end{lstlisting}
\end{exampleblock}

{\color{red} $\times$ Erreur} apply(TB) in FunBA clashes with apply(TB) in FunBB \\
\hspace{4em}return type TA is not compatible with TB

\end{frame}

\begin{frame}[fragile]{Les $\lambda$-Expressions: Définition Syntaxique}

La syntaxe adoptée est fortement inspirée de C\# et Scala 

\vspace{1em}

\begin{block}{}
\begin{lstlisting}
   Lambda  ::= TypeParameters? ArgList "->" Body

   Body    ::= Expression
             | "{" [ Statement ";" ]* "}"

   ArgList ::= "(" ")"
             | "(" Type Identifier [ "," Type Identifier ]* ")"
\end{lstlisting}

\breakinframe

\vspace{-1em}

\begin{lstlisting}
             | Identifier
             | "(" Identifier [ "," Identifier ]* ")"
\end{lstlisting}
\end{block}

\vspace{1em}

\begin{alertblock}{}
  \coffeebreak ~ L'inférence de type devient une clef de voute du langage
\end{alertblock}

\end{frame}

\begin{frame}[fragile]{Variable Effectivement Finale}

\begin{block}{Définition (Variable Effectivement Finale)}
Une variable est dite effectivement finale si dans le bloc de base 
qui la porte celle ci est initialisée et jamais modifiée.
\end{block}

\vspace{1em}

Immutabilité des variables capturées dans les $\lambda$-Expressions:
\begin{enumerate}
\item Déclarée par l'usage du mot clef {\tt final} ou
\item Vérifiée par le compilateur
\end{enumerate}

\vspace{1em}

\begin{alertblock}{}
  \coffeebreak ~ Variable effectivement finale uniquement pour les $\lambda$-Expressions !
\end{alertblock}

\end{frame}

\begin{frame}[fragile]{Les Références de Méthodes et Constructeurs}

\begin{block}{Définition (Référence de Méthodes et Constructeurs)}
  Une réference de méthode (resp. de constructeur) est traitée de
  façon similaire à une $\lambda$-Expression mais elle permet de
  manipuler une méthode (resp.un constructeur) déjà existante.
\end{block}

\vspace{1em}

Syntaxe étendue aux références:

\begin{block}{}
\begin{lstlisting}
   Expression ::= ...
                | Expression "::" methodName
                | ClassType "::" methodName
                | ClassType "::" new                
\end{lstlisting}
\end{block}

\vspace{1em}

\end{frame}

\begin{frame}[fragile]{Exemples de Références}

\begin{exampleblock}{}
\begin{lstlisting}
  interface Fun0<R> { R apply(); }
  interface Fun1<E,R> { R apply(E e); }

  public class Cell<C> {
    private final C value;   
    public Cell(C value) { this.value = value; }
    public boolean isNull() { return this.value == null; }
  }
\end{lstlisting}
\end{exampleblock}

\begin{exampleblock}{}
\begin{lstlisting}
    Fun1<Integer,Cell<Integer>> factory1 = i -> new Cell<>(i);
\end{lstlisting}
\vspace{-1em}\begin{lstlisting}
    Fun1<Integer,Cell<Integer>> factory2 = Cell<Integer>::new;
\end{lstlisting}
\begin{lstlisting}
    Fun1<Cell<Integer>,Boolean> isNullC1 = c -> c.isNull();
\end{lstlisting}
\vspace{-1em}\begin{lstlisting}
    Fun1<Cell<Integer>,Boolean> isNullC2 = Cell<Integer>::isNull;
\end{lstlisting}
\begin{lstlisting}
    Cell<Integer> cell = factory1.apply(1);
    Fun0<Boolean> isNullO1 = () -> cell.isNull();
\end{lstlisting}
\vspace{-1em}\begin{lstlisting}
    Fun0<Boolean> isNullO2 = cell::isNull;
\end{lstlisting}
\end{exampleblock}

\end{frame}

\begin{frame}{A propos de la compilation}

$\lambda$-Expression vs. classe interne anonyme:
\begin{enumerate}
\item Capture de {\tt this} excepté dans une méthode {\tt static}.
\item Compilation équivalente en présence de variables finales
\begin{itemize}
\item Constant pour les types primitifs ou 
\item Constant Pool si c'est une chaîne de caractères
\item Passage au constructeur sinon
\end{itemize}
\item Petite variation en présence de variables effectivement finales
\begin{itemize}
\item Passage au constructeur uniquement
\end{itemize}
\end{enumerate}

\vspace{1em}
$\lambda$-Expression uniquement:
\begin{enumerate}
\item Lambda Lifting par transcription objet
\item Pas de gestion de la récursion terminale
\begin{itemize}
\item Stack overflow
\end{itemize}
\end{enumerate}

\end{frame}

\begin{frame}[fragile]{Extension Virtuelle et Héritage multiple}

Héritage multiple implique de potentiels conflits lors de la résolution:
\begin{itemize}
\item Définition des classes tout d'abord, les interfaces ensuite
\item Méthodes surchargées ignorées
\item Conflit réglé par le développeur
\end{itemize}

\begin{exampleblock}{}
\begin{lstlisting}
  interface D extends A { int calc(int i) default { return i; } }

  class E implements B, D { public int getIncr() { retun 10; } }
\end{lstlisting}
\end{exampleblock}
{\color{red} $\times$ Erreur} class E inherits unrelated defaults for calc(int) from types B and D

\begin{exampleblock}{}
\begin{lstlisting}
  class E implements B, D {
    public int getIncr() { retun 10; }
    public int calc(int i) { return D.super.calc(i); }
  }
\end{lstlisting}
\end{exampleblock}
\end{frame}

  public final class Runtime {
    private final PrintStream out;
    private final InputStream in;

    public Runtime(PrintStream out, InputStream in, byte[] memory, int cursor) {
      this.out = out; this.in = in;
      this.memory = memory; this.cursor = cursor;
    }

    private Runtime set(int v) { memory[cursor] = (byte)v; return this; }
    public boolean isZero() { return memory[cursor] == 0; }
    public Runtime right() { return new Runtime(out, in, memory, cursor + 1); }
    public Runtime left() { return new Runtime(out, in, memory, cursor - 1); }
    public Runtime incr() { return this.set(memory[cursor] + 1); }
    public Runtime decr() { return this.set(memory[cursor] - 1); }
    public Runtime write() { out.print((char)memory[cursor]); return this;}
    public Runtime read() throws IOException { return this.set(in.read()); };
  }

        Runtime currentRuntime = env;
        while(!currentRuntime.isZero()) {
          currentRuntime = execute(currentRuntime);
        }
        return currentRuntime;

  public static BrainFun compile(byte[] program) {
    final Stack<BrainFun> code = new Stack<>();
    code.push(e -> e);

    for(byte instruction: program) {
      switch (instruction) {
        case '>': code.push(code.pop().seq(Runtime::right)); break;
        case '<': code.push(code.pop().seq(Runtime::left)); break;
        case '+': code.push(code.pop().seq(Runtime::incr)); break;
        case '-': code.push(code.pop().seq(Runtime::decr)); break;
        case '.': code.push(code.pop().seq(Runtime::write)); break;
        case ',': code.push(code.pop().seq(Runtime::read)); break;
        case '[': code.push(e -> e); break;
        case ']': final BrainFun loop = code.pop().loop();
                  code.push(code.pop().seq(loop)); break;
        default: // Skip
      }
    }

    return code.pop();
